```typescript
// 需求复杂度评估器
interface ComplexityAssessor {
  assess(input: string): ComplexityResult {
    // 评估维度
    const dimensions = {
      // 业务逻辑复杂度 (0-100)
      businessLogic: this.analyzeBusinessLogic(input),
      
      // 涉及角色数量 (0-100)
      stakeholderCount: this.countStakeholders(input),
      
      // 流程分支数量 (0-100)
      branchComplexity: this.analyzeBranches(input),
      
      // 合规要求强度 (0-100)
      complianceIntensity: this.checkCompliance(input),
      
      // 创新程度 (0-100, 越高越需要人工介入)
      innovationDegree: this.assessInnovation(input)
    };
    
    // 综合评分
    const overallScore = weightedAverage(dimensions, {
      businessLogic: 0.25,
      stakeholderCount: 0.20,
      branchComplexity: 0.25,
      complianceIntensity: 0.20,
      innovationDegree: 0.10
    });
    
    // 分级
    if (overallScore < 30) return { level: 'simple', strategy: 'single-round' };
    if (overallScore < 60) return { level: 'medium', strategy: 'multi-round' };
    if (overallScore < 80) return { level: 'complex', strategy: 'step-by-step' };
    return { level: 'very-complex', strategy: 'human-first' };
  }
}
```

```typescript
interface ConversationCompressor {
```

```python
        }
        
        # 4. 存入向量库
        await self.vector_store.upsert(
            ids=[f"{prd.id}_{i}" for i in range(len(chunks))],
            vectors=embeddings,
            metadata=metadata

    # PRD分块策略
    def chunk_prd(self, prd: PRDDocument) -> List[Chunk]:

                ))
        
        # 提取关键段落单独索引
        key_sections = self.extract_key_sections(prd)
```

```typescript
// 多路召回检索器
interface MultiRetriever {
  async retrieve(query: string, context: QueryContext): Promise<RetrievalResult> {
    // 并行执行多种检索
    const [
      semanticResults,
      keywordResults,
      templateResults,
      historyResults
    ] = await Promise.all([
      this.semanticSearch(query, context),
      this.keywordSearch(query, context),
      this.templateMatch(query, context),
      this.historySearch(query, context)
    ]);
    
    // 重排序
    const reranked = await this.rerank(
      [semanticResults, keywordResults, templateResults, historyResults],
      query,
      context
    );
    
    // 去重和组装
    return this.deduplicateAndAssemble(reranked);
  }
  
  // 语义检索（向量相似度）
  async semanticSearch(query: string, context: QueryContext): Promise<Result[]> {
    const embedding = await this.getEmbedding(query);
    
    return this.vectorStore.query({
      vector: embedding,
      topK: 10,
      filter: {
        industry: context.industry,
        quality_score: { $gte: 70 }  // 只检索高质量PRD
      }
    });
  }
  
  // 关键词检索
  async keywordSearch(query: string, context: QueryContext): Promise<Result[]> {
    const keywords = this.extractKeywords(query);
    
    return this.elasticSearch.search({
      index: 'prd_documents',
      query: {
        bool: {
          should: keywords.map(k => ({
            multi_match: {
              query: k,
              fields: ['title^3', 'content', 'tags^2'],
              fuzziness: 'AUTO'
            }
          }))
        }
      },
      size: 10
    });
  }
  
  // 模板匹配
  async templateMatch(query: string, context: QueryContext): Promise<Result[]> {
    const templates = await this.getTemplates(context.industry);
    
    // 计算查询与模板的相似度
    return templates
      .map(t => ({
        ...t,
        similarity: this.calculateSimilarity(query, t.description)
      }))
      .sort((a, b) => b.similarity - a.similarity)
      .slice(0, 3);
  }
  
  // 用户历史检索
  async historySearch(query: string, context: QueryContext): Promise<Result[]> {
    // 检索用户自己写过的类似PRD
    return this.vectorStore.query({
      vector: await this.getEmbedding(query),
      topK: 5,
      filter: {
        created_by: context.userId,
        project_id: context.projectId
      }
    });
  }
  
  // 重排序模型
  async rerank(results: Result[][], query: string, context: QueryContext): Promise<Result[]> {
    const allResults = results.flat();
    
    // 使用Cross-Encoder进行精确重排序
    const pairs = allResults.map(r => [query, r.content]);
    const scores = await this.crossEncoder.score(pairs);
    
    // 综合考虑多维度因素
    return allResults
      .map((r, i) => ({
        ...r,
        finalScore: this.computeFinalScore(r, scores[i], context)
      }))
      .sort((a, b) => b.finalScore - a.finalScore);
  }
  
  computeFinalScore(result: Result, semanticScore: number, context: QueryContext): number {
    return (
      semanticScore * 0.35 +                    // 语义相关性
      result.keywordMatchScore * 0.25 +         // 关键词匹配
      result.qualityScore * 0.20 +              // 质量分数
      (result.userId === context.userId ? 0.15 : 0) +  // 用户偏好
      result.recencyScore * 0.05                // 时效性
    );
  }
}
```

```typescript
// MVP自动规划器
interface MVPPlanner {
  planMVP(features: Feature[]): MVPPlan {
    // 按价值/成本比排序
    const scoredFeatures = features.map(f => ({
      ...f,
      valueScore: this.assessValue(f),
      costScore: this.assessCost(f),
      ratio: this.assessValue(f) / this.assessCost(f)
    }));
    
    const sorted = scoredFeatures.sort((a, b) => b.ratio - a.ratio);
    
    // 选取前60%价值的功能作为MVP
    const totalValue = sum(features.map(f => f.valueScore));
    let accumulatedValue = 0;
    const mvpFeatures = [];
    
    for (const f of sorted) {
      if (accumulatedValue / totalValue < 0.6) {
        mvpFeatures.push(f);
        accumulatedValue += f.valueScore;
      }
    }
    
    return {
      mvpFeatures,
      phasedRoadmap: this.createPhases(sorted, mvpFeatures),
      rationale: this.generateRationale(mvpFeatures),
      riskAnalysis: this.analyzeRisks(mvpFeatures)
    };
  }
}
```
